This function turns the string "true" or "false" into a boolean

Code: 
function stringToBool(string) {
    if(string === "true") {
        return true;
    }
    else if(string === "false") {
        return false;
    }
    else {
        return undefined;
    }
}

This function removes all "e"s from the input

Code: 
function removeE(string) {
    output = "";
    for(i=0; i<string.length; i++) {
        if(string[i] != "e" && string[i] != "E") {
            output += string[i];
        }
    }
    return output;
}

This function removes all of an input letter from the input

Code: 
function removeLetter(letter, string) {
    output = "";
    for(i=0; i<string.length; i++) {
        if(string[i] != letter.toLowerCase() && string[i] != letter.toUpperCase()) {
            output += string[i];
        }
    }
    return output;

My use of class extend


function isVowel(letter) {
    return ['a', 'e', 'i', 'o', 'u'].indexOf(letter.toLowerCase()) !== -1
}
class animal {
    constructor(name, age, type) {
        this.name = name;
        this.age = age;
        this.type = type;
        if(type === undefined) {
            this.type = 'animal';
        }
    }
    animalType() {
        this.a = 'a'
        if(isVowel(this.type[0])) {
            this.a = 'an'
        }
        if(this.age === undefined) {
            return `${this.name} is ${this.a} ${this.type}.`
        }
        else{
            return `${this.name} is ${this.a} ${this.type} that is ${this.age} years old.`
        }
    }
    talk() {
        return this.animalType()
    }
}
class cat extends animal {
    constructor(name, age) {
        super(name, age, 'cat')
    }
    talk() {
        return `${this.animalType()} ${this.name} goes meow.`
    }                                                                   
}
class dog extends animal {
    constructor(name, age) {
        super(name, age, 'dog')
    }
    talk() {
        return `${this.animalType()} ${this.name} goes woof.`
    }                                                                   
}
